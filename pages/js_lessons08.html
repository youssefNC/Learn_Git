day = day.trim().charAt(0).toUpperCase()+day.trim().slice(1).toLowerCase();
console.log(day);

switch (day) {
    case "Friday":
    case "Saturday":
    case "Sunday":
    console.log(`No Appointments Available On ${day}`);
    break;
    case "Monday":
    case "Thursday":
    console.log(`Appointments available on ${day} From 10:00 AM To 5:00 PM`);
    break;
    case "Tuesday":
    console.log(`Appointments available on ${day} From 11:00 AM To 6:00 PM`);
    break;
    case "Wednesday":
    console.log(`Appointments available on ${day} From 12:00 AM To 7:00 PM`);
    break;
    default:
        console.log(`${day} Its Not A Valid Day`);
}
Switch And If Challange
رابط فيديو التحدي على قناة الزيرو هنا 

let job = "Manager";
let salary = 0;

switch (job) {
    case "Manager":
    salary = 8000;
    console.log(`Manager salary is ${salary}`);
    break;
    case "IT":
    case "Support":
    salary = 6000;
    console.log(`IT and Support: salary is ${salary}`);
    break;
    case "Developer":
    case "Designer":
    salary = 7000;
    console.log(`Developer and Designer: salary is ${salary}`);
    break;
    default:
    console.log("Unknown salary");
    break;
}
